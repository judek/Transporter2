    In case you want to load data synchronously into an image here is the code (javascript):
            var image = new Image(imageProps.Width, imageProps.Height);        var sjax = new XMLHttpRequest();        sjax.open("GET", imageProps.ImageURL, false);        sjax.overrideMimeType("text/plain; charset=x-user-defined");        sjax.send(null);        BlobBuilder = window.MozBlobBuilder || window.WebKitBlobBuilder || window.BlobBuilder;        var byteArray = new Uint8Array(sjax.responseText.length);        for (var i = 0; i < sjax.responseText.length; i++) {            byteArray[i] = sjax.responseText.charCodeAt(i) & 0xff;        }        var bb = new BlobBuilder();        bb.append(byteArray.buffer);        window.URL = window.URL || window.webkitURL;        image.src = window.URL.createObjectURL(bb.getBlob('image/png'));
